``` r
plot_df = tibble(
  x = rnorm(1000, mean = 0, sd = 1),
  y = rnorm(1000, mean = 0, sd = 1),
  vector_logical = c(x + y > 0),
  typeof(vector_logical),
  median(x),
  mean(x),
  n = as.numeric(vector_logical),
  f = as.factor(vector_logical)
)
ggplot(plot_df, aes(x = x, y = y, col = "red" )) + geom_point()
```
